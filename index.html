  <title>Мой каталог скейт-дек</title>

  <script src="https://cdn.tailwindcss.com"></script>

  <style>

    body { overscroll-behavior: none; }

    .flip-card-inner { transition: transform 0.7s cubic-bezier(0.68,-0.55,0.265,1.55); transform-style: preserve-3d; }

    .flip-card.flipped .flip-card-inner { transform: rotateY(180deg); }

    .flip-card-front, .flip-card-back { backface-visibility: hidden; }

    .flip-card-back { transform: rotateY(180deg); }

  </style>

<body class="min-h-screen bg-black text-white">

  <!-- Главный экран — просмотр -->

  <div id="viewer" class="relative h-screen flex flex-col">

    <div id="gallery" class="flex-1 flex items-center justify-center"></div>

   

    <!-- Кнопки навигации -->

    <button id="prev" class="fixed left-4 top-1/2 -translate-y-1/2 z-40 text-6xl opacity-70 hover:opacity-100">←</button>

    <button id="next" class="fixed right-4 top-1/2 -translate-y-1/2 z-40 text-6xl opacity-70 hover:opacity-100">→</button>

   

    <!-- Нижняя панель -->

    <div class="fixed bottom-0 left-0 right-0 bg-black/90 backdrop-blur p-4 flex justify-between items-center z-50">

      <span id="counter" class="text-xl font-bold">1 / 1</span>

      <button onclick="openEditor()" class="bg-purple-600 hover:bg-purple-700 px-6 py-3 rounded-full font-bold text-lg">

        ✎ Редактировать каталог

      </button>

    </div>

  </div>

  <!-- Экран редактора -->

  <div id="editor" class="hidden fixed inset-0 bg-black z-50 overflow-y-auto">

    <div class="p-6 max-w-4xl mx-auto">

      <div class="flex justify-between items-center mb-8">

        <h1 class="text-4xl font-bold">Редактирование каталога</h1>

        <button onclick="closeEditor()" class="text-4xl">×</button>

      </div>

      <button onclick="addNewDeck()" class="mb-8 bg-green-600 hover:bg-green-700 px-8 py-4 rounded-xl text-xl font-bold">

        + Добавить новую деку

      </button>

      <div id="deckList"></div>

    </div>

  </div>

  <script>

    let decks = JSON.parse(localStorage.getItem('skateDecks')) || [];

    let current = 0;

    const viewer = document.getElementById('viewer');

    const editor = document.getElementById('editor');

    const gallery = document.getElementById('gallery');

    const counter = document.getElementById('counter');

    // Автозаполнение, если база пустая (пример)

    if (decks.length === 0) {

      // Попробуем подтянуть все картинки из папки decks/

      for (let i = 1; i <= 500; i++) {

        ['jpg','jpeg','png','webp','gif'].forEach(ext => {

          const url = `decks/\( {i}. \){ext}`;

          const img = new Image();

          img.onload = () => {

            if (!decks.some(d => d.image === url)) {

              decks.push({

                image: url,

                name: `Дека ${i}`,

                params: [

                  {key: "Бренд", value: ""},

                  {key: "Райдер", value: ""},

                  {key: "Год", value: ""},

                  {key: "Серия", value: ""},

                  {key: "", value: ""},

                  {key: "", value: ""},

                  {key: "", value: ""},

                  {key: "", value: ""},

                  {key: "", value: ""},

                  {key: "", value: ""}

                ]

              });

              saveAndRender();

            }

          };

          img.src = url;

        });

      }

    }

    function saveAndRender() {

      localStorage.setItem('skateDecks', JSON.stringify(decks));

      renderViewer();

      renderEditorList();

    }

    function renderViewer() {

      if (decks.length === 0) {

        gallery.innerHTML = `<div class="text-3xl text-center">Каталог пуст<br><br>Нажми «Редактировать каталог» и добавь деки</div>`;

        counter.textContent = '0 / 0';

        return;

      }

      counter.textContent = `\( {current+1} / \){decks.length}`;

      const deck = decks[current];

      const name = deck.name || deck.image.split('/').pop();

      gallery.innerHTML = `

        <div class="flip-card w-full h-full cursor-pointer" onclick="this.classList.toggle('flipped')">

          <div class="flip-card-inner w-full h-full">

            <div class="flip-card-front absolute inset-0 flex items-center justify-center bg-black p-8">

              <img src="${deck.image}" class="max-w-full max-h-full object-contain drop-shadow-2xl">

            </div>

            <div class="flip-card-back absolute inset-0 bg-gradient-to-br from-zinc-900 to-black p-8 overflow-y-auto">

              <h2 class="text-4xl font-bold mb-8 text-center">${name}</h2>

              \( {deck.params.map(p => p.key ? `<div class="mb-4"><span class="font-bold text-purple-400"> \){p.key}:</span> ${p.value}</div>` : '').join('')}

            </div>

          </div>

        </div>

      `;

    }

    document.getElementById('prev').onclick = () => { current = (current - 1 + decks.length) % decks.length; renderViewer(); };

    document.getElementById('next').onclick = () => { current = (current + 1) % decks.length; renderViewer(); };

    function openEditor() { viewer.classList.add('hidden'); editor.classList.remove('hidden'); renderEditorList(); }

    function closeEditor() { editor.classList.add('hidden'); viewer.classList.remove('hidden'); renderViewer(); }

    function renderEditorList() {

      const list = document.getElementById('deckList');

      list.innerHTML = decks.map((deck, i) => `

        <div class="bg-zinc-900 rounded-xl p-6 mb-6 border ${i===current?'border-purple-500':'border-zinc-700'}">

          <div class="flex justify-between items-start mb-4">

            <input type="text" value="${deck.name||''}" placeholder="Название деки" class="bg-transparent text-2xl font-bold w-full mr-4"

                   onchange="decks[${i}].name=this.value; saveAndRender()">

            <button onclick="deleteDeck(${i})" class="text-red-500 text-3xl">×</button>

          </div>

          <img src="${deck.image}" class="w-48 h-auto rounded-lg mb-4 object-contain bg-black">

          <div class="grid grid-cols-1 md:grid-cols-2 gap-3">

            ${deck.params.map((p,j) => `

              <div class="flex gap-2">

                <input placeholder="Параметр" value="${p.key}" class="bg-zinc-800 px-3 py-2 rounded flex-1"

                       onchange="decks[\( {i}].params[ \){j}].key=this.value; saveAndRender()">

                <input placeholder="Значение" value="${p.value}" class="bg-zinc-800 px-3 py-2 rounded flex-1"

                       onchange="decks[\( {i}].params[ \){j}].value=this.value; saveAndRender()">

              </div>

            `).join('')}

          </div>

        </div>

      `).join('') || '<p class="text-center text-xl opacity-50">Пока нет ни одной деки</p>';

    }

    function addNewDeck() {

      const newDeck = {

        image: `decks/${decks.length+1}.jpg`, // подправь, если нужно

        name: "Новая дека",

        params: Array(10).fill().map(() => ({key: "", value: ""}))

      };

      decks.push(newDeck);

      current = decks.length - 1;

      saveAndRender();

      renderEditorList();

    }

    function deleteDeck(i) {

      if (confirm('Точно удалить эту деку?')) {

        decks.splice(i, 1);

        if (current >= decks.length) current = 0;

        saveAndRender();

      }

    }

    // Свайпы и клавиши

    let sx = 0;

    gallery.addEventListener('touchstart', e => sx = e.touches[0].clientX);

    gallery.addEventListener('touchend', e => {

      const dx = e.changedTouches[0].clientX - sx;

      if (Math.abs(dx) > 70) dx > 0 ? document.getElementById('prev').click() : document.getElementById('next').click();

    });

    window.addEventListener('keydown', e => {

      if (e.key === 'ArrowLeft') document.getElementById('prev').click();

      if (e.key === 'ArrowRight') document.getElementById('next').click();

    });

    // Старт

    renderViewer();

  </script>

</body>

</html>
